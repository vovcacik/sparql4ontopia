To install this plugin copy these files to your Ontopia installation:

.\jar\sparql4ontopia.jar   ->    %ONTOPIA_HOME%\apache-tomcat\common\lib\
.\lib\*.*                  ->    %ONTOPIA_HOME%\apache-tomcat\common\lib\
.\jsp\*.*                  ->    %ONTOPIA_HOME%\apache-tomcat\webapps\omnigator\plugins\query\
